# 04. Rotate array by K places
> Solution of this problem
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

void rotateRight(vector<int>& a, int k) {
    int n = a.size();
    if (n == 0) return;

    k = k % n;
    if (k < 0) k += n;

    reverse(a.begin(), a.end());
    reverse(a.begin(), a.begin() + k);
    reverse(a.begin() + k, a.end());
}

int main() {
    vector<int> a = {1,2,3,4,5,6,7};
    rotateRight(a, 3);

    for (int x : a) cout << x << " ";
}
```
