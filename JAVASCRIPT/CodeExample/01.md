# 01. Find a minimum and maximum element in array
> solution of this problem
```cpp
#include <iostream>
#include <vector>
using namespace std;

pair<int, int> findMinMax(const vector<int>& a){
  int mn = a[0];
  int mx = a[0];

  for (int x : a){
    if(x < mn) mn = x;
    if(x > mx) mx = x;
  }
  return {mn, mx};
}

int main (){
  vector<int> = {5,8,9,6,31,7,0};
  auto ans = finMinMax(a);

  cout << "Min:" << ans.first << endl;
  cout << "Max:" << ans.second << endl;
}
```
