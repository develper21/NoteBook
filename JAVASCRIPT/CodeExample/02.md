# 02. Find the second in array
> Solution Of this Problem
```cpp
#include <iostream>
#include <vector>
#include <climits>
using namespace std;

int secondLargest(const vector<int>& a) {
    int first = INT_MIN;
    int second = INT_MIN;

    for (int x : a) {
        if (x > first) {
            second = first;
            first = x;
        }
        else if (x < first && x > second) {
            second = x;
        }
    }
    return second == INT_MIN ? -1 : second;
}

int main() {
    vector<int> arr = {5, 9, 1, 9, 7};

    int ans = secondLargest(arr);
    cout << "Second Largest = " << ans << endl;
    return 0;
}
```
