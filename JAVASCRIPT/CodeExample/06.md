# 06. Remove Dublicate Array in Sorted & Unsorted
> 6A sorted Array problem solution
```cpp
#include <iostream>
#include <vector>
using namespace std;

int removeDupSorted(vector<int>& a) {
    if (a.empty()) return 0;

    int write = 1;
    for (int i = 1; i < a.size(); i++) {
        if (a[i] != a[i - 1])
            a[write++] = a[i];
    }
    a.resize(write);
    return write;
}

int main() {
    vector<int> a = {1,1,2,2,3,3};
    removeDupSorted(a);

    for (int x : a) cout << x << " ";
}
```
---
## 6A Complexity
> Time Complexity
```cpp
// Array scan in only one pass
for (int i = 1; i < a.size(); i++)
// so Time Complexity is : o(n)
```

> Space Complexity
```cpp
resize() is O(1) amortized (just changes size metadata)
```

> 6B Unsorted Array problem solution
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int removeDupUnsorted(vector<int>& a) {
    sort(a.begin(), a.end());

    auto it = unique(a.begin(), a.end());
    a.erase(it, a.end());

    return a.size();
}

int main() {
    vector<int> a = {3,1,2,3,2,4,1};
    removeDupUnsorted(a);

    for (int x : a) cout << x << " ";
}

```
---
## 6B Complexity
> Time Complexity
```cpp
// Sorting:
sort(a.begin(), a.end());

// unique():
auto it = unique(a.begin(), a.end());

// erase():
a.erase(it, a.end());


// Final Coplexity:
O(n log n) + O(n) + O(n) = O(n log n)
```

> Space Complexity
```cpp
Space Complexity: O(1) (sort + unique both in-place)
```
